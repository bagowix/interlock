1"""Redis-backed shared state for coordinated, distributed breakers. 

2 

3Optional extra ``interlock-cb[redis]`` — not re-exported from the package root; 

4the core stays zero-dependency. ``RedisStorage`` (sync) and ``AsyncRedisStorage`` 

5(async) implement the ``Storage`` / ``AsyncStorage`` contracts over redis-py. 

6 

7All state for one breaker lives in a single Redis hash keyed 

8``<prefix><name>``. Every transition runs as a Lua script so racing instances 

9stay consistent: the script is the atomic critical section. Lua carries 

10mechanism only — atomic CAS, server-time elapsed checks (``TIME``, since instance 

11clocks are not comparable), probe leasing and counters, and TTL refresh. 

12Threshold *policy* stays in the core state machine, which calls ``trip_open`` / 

13``close`` once it has decided. 

14 

15Wire-compatible with Redis, Valkey, and any RESP server: this speaks plain 

16commands and ``EVAL``, with no server-specific features. The scripts call 

17``TIME`` before writing, so the server must replicate scripts by effects — 

18Redis >= 5.0 or any Valkey (the ``redis>=5.0.0`` dependency pin is the 

19*client* library version, not the server's). 

20""" 

21 

22from typing import Any 

23 

24import redis 

25import redis.asyncio 

26 

27from interlock.outcome import Outcome 

28from interlock.shared import ProbeLease, SharedState 

29from interlock.state import State

31__all__ = ('AsyncRedisStorage', 'RedisStorage')

33_DEFAULT_PREFIX = 'interlock:cb:' 

34 

35# Hash fields, in the fixed order every script returns them. 

36_FIELDS = ( 

37 "'state','opened_at','version'," 

38 "'probes_permitted','probes_remaining','probes_completed'," 

39 "'probe_failures','probe_slows'" 

40) 

41_RETURN_STATE = f"return redis.call('HMGET', KEYS[1], {_FIELDS})" 

42 

43# opened_at is stored as integer microseconds (epoch, from the server's TIME), 

44# kept under 2^53 so it round-trips exactly through Lua's double numbers. 

45_NOW_US = ( 

46 "(function() local t = redis.call('TIME') " 

47 'return tonumber(t[1]) * 1000000 + tonumber(t[2]) end)()' 

48) 

49 

50# expected_version < 0 means unfenced; a fenced-out call performs no write at 

51# all (not even a TTL refresh — a stale actor must not extend the key's life). 

52_TRIP_OPEN = f""" 

53local key = KEYS[1] 

54local ttl_ms = tonumber(ARGV[1]) 

55local expected = tonumber(ARGV[2]) 

56local version = tonumber(redis.call('HGET', key, 'version') or '0') 

57if expected < 0 or version == expected then 

58 if redis.call('HGET', key, 'state') ~= 'open' then 

59 redis.call('HSET', key, 

60 'state', 'open', 'opened_at', {_NOW_US}, 'version', version + 1, 

61 'probes_permitted', 0, 'probes_remaining', 0, 'probes_completed', 0, 

62 'probe_failures', 0, 'probe_slows', 0, 'v', 1) 

63 end 

64 redis.call('PEXPIRE', key, ttl_ms) 

65end 

66{_RETURN_STATE} 

67""" 

68 

69_BEGIN_HALF_OPEN = f""" 

70local key = KEYS[1] 

71local wait_us = tonumber(ARGV[1]) * 1000000 

72local permitted = tonumber(ARGV[2]) 

73local ttl_ms = tonumber(ARGV[3]) 

74if redis.call('HGET', key, 'state') == 'open' then 

75 local opened_at = tonumber(redis.call('HGET', key, 'opened_at') or '0') 

76 if {_NOW_US} - opened_at >= wait_us then 

77 local version = tonumber(redis.call('HGET', key, 'version') or '0') + 1 

78 redis.call('HSET', key, 

79 'state', 'half_open', 'version', version, 

80 'probes_permitted', permitted, 'probes_remaining', permitted, 

81 'probes_completed', 0, 'probe_failures', 0, 'probe_slows', 0) 

82 end 

83end 

84redis.call('PEXPIRE', key, ttl_ms) 

85{_RETURN_STATE} 

86""" 

87 

88_LEASE_PROBE = f""" 

89local key = KEYS[1] 

90local ttl_ms = tonumber(ARGV[1]) 

91local granted = 0 

92if redis.call('HGET', key, 'state') == 'half_open' then 

93 if tonumber(redis.call('HGET', key, 'probes_remaining') or '0') > 0 then 

94 redis.call('HINCRBY', key, 'probes_remaining', -1) 

95 redis.call('HINCRBY', key, 'version', 1) 

96 granted = 1 

97 end 

98end 

99redis.call('PEXPIRE', key, ttl_ms) 

100local r = redis.call('HMGET', KEYS[1], {_FIELDS}) 

101table.insert(r, 1, granted) 

102return r 

103""" 

104 

105# Tallies only while HALF_OPEN: a probe outcome arriving after the state moved 

106# on (another instance tripped or closed) must not pollute the new accounting. 

107_RECORD_PROBE = f""" 

108local key = KEYS[1] 

109local is_failure = tonumber(ARGV[1]) 

110local is_slow = tonumber(ARGV[2]) 

111local ttl_ms = tonumber(ARGV[3]) 

112if redis.call('HGET', key, 'state') == 'half_open' then 

113 redis.call('HINCRBY', key, 'probes_completed', 1) 

114 redis.call('HINCRBY', key, 'version', 1) 

115 if is_failure == 1 then redis.call('HINCRBY', key, 'probe_failures', 1) end 

116 if is_slow == 1 then redis.call('HINCRBY', key, 'probe_slows', 1) end 

117 redis.call('PEXPIRE', key, ttl_ms) 

118end 

119{_RETURN_STATE} 

120""" 

121 

122_CLOSE = f""" 

123local key = KEYS[1] 

124local ttl_ms = tonumber(ARGV[1]) 

125local expected = tonumber(ARGV[2]) 

126local version = tonumber(redis.call('HGET', key, 'version') or '0') 

127if expected < 0 or version == expected then 

128 redis.call('HSET', key, 

129 'state', 'closed', 'opened_at', 0, 'version', version + 1, 

130 'probes_permitted', 0, 'probes_remaining', 0, 'probes_completed', 0, 

131 'probe_failures', 0, 'probe_slows', 0, 'v', 1) 

132 redis.call('PEXPIRE', key, ttl_ms) 

133end 

134{_RETURN_STATE} 

135"""

138def _s(value: object) -> str: 

139 """Decode a Redis reply scalar (bytes or str) to ``str``.""" 

140 return value.decode() if isinstance(value, bytes) else str(value) 

141 

142 

143def _ms(ttl: float) -> int: 

144 """Convert a TTL in seconds to whole milliseconds for ``PEXPIRE``. 

145 

146 Rejects TTLs under one millisecond: ``PEXPIRE key 0`` would delete the key 

147 outright instead of expiring it. 

148 """ 

149 ms = int(ttl * 1000) 

150 if ms <= 0: 

151 msg = f'ttl must be at least 0.001 seconds, got {ttl!r}' 

152 raise ValueError(msg) 

153 return ms 

154 

155 

156def _fence(expected_version: int | None) -> int: 

157 """Encode ``expected_version`` for Lua: ``-1`` disables the fence.""" 

158 return -1 if expected_version is None else expected_version

161def _shared_from_values(values: list[Any]) -> SharedState: 

162 """Build a ``SharedState`` from an ``HMGET`` reply in ``_FIELDS`` order.""" 

163 if not values or values[0] is None: 

164 return SharedState.closed() 

165 

166 return SharedState( 

167 state=State(_s(values[0])), 

168 opened_at=int(_s(values[1])) / 1_000_000, 

169 version=int(_s(values[2])), 

170 probes_permitted=int(_s(values[3])), 

171 probes_remaining=int(_s(values[4])), 

172 probes_completed=int(_s(values[5])), 

173 probe_failures=int(_s(values[6])), 

174 probe_slows=int(_s(values[7])), 

175 ) 

176 

177 

178def _shared_from_map(mapping: dict[Any, Any]) -> SharedState | None: 

179 """Build a ``SharedState`` from an ``HGETALL`` reply, or ``None`` if absent.""" 

180 if not mapping: 

181 return None 

182 

183 fields = {_s(k): v for k, v in mapping.items()} 

184 order = ( 

185 'state', 

186 'opened_at', 

187 'version', 

188 'probes_permitted', 

189 'probes_remaining', 

190 'probes_completed', 

191 'probe_failures', 

192 'probe_slows', 

193 ) 

194 return _shared_from_values([fields.get(name) for name in order]) 

195 

196 

197def _positive(name: str, value: float) -> float: 

198 if value <= 0: 

199 msg = f'{name} must be positive, got {value!r}' 

200 raise ValueError(msg) 

201 return value

204class RedisStorage: 

205 """Synchronous ``Storage`` backed by a ``redis.Redis`` client. 

206 

207 The three keyword floats are the coordination tuning knobs the engine reads 

208 off the storage object: how long state keys live without a refresh 

209 (``state_ttl``), how often the cached shared view is refreshed 

210 (``poll_interval``), and how long the breaker runs purely locally after a 

211 storage failure before retrying (``retry_backoff``). 

212 

213 Args: 

214 client: A connected ``redis.Redis`` instance. 

215 key_prefix: Namespace prepended to every breaker name. 

216 state_ttl: Key lifetime in seconds; refreshed on every write. 

217 poll_interval: Seconds between background view refreshes. 

218 retry_backoff: Seconds to stay degraded after a storage failure. 

219 """ 

220 

221 def __init__( 

222 self, 

223 client: redis.Redis, 

224 *, 

225 key_prefix: str = _DEFAULT_PREFIX, 

226 state_ttl: float = 300.0, 

227 poll_interval: float = 1.0, 

228 retry_backoff: float = 5.0, 

229 ) -> None: 

230 self._client = client 

231 self._prefix = key_prefix 

232 self.state_ttl = _positive('state_ttl', state_ttl) 

233 self.poll_interval = _positive('poll_interval', poll_interval) 

234 self.retry_backoff = _positive('retry_backoff', retry_backoff) 

235 

236 def _key(self, name: str) -> str: 

237 return f'{self._prefix}{name}' 

238 

239 def read(self, name: str) -> SharedState | None: 

240 """Return the current shared view, or ``None`` if no key exists.""" 

241 return _shared_from_map(self._client.hgetall(self._key(name))) 

242 

243 def trip_open( 

244 self, *, name: str, ttl: float, expected_version: int | None = None 

245 ) -> SharedState: 

246 """Atomically transition to ``OPEN`` (idempotent while already open). 

247 

248 With ``expected_version``, fenced: a no-op unless the backend still 

249 holds that version. 

250 """ 

251 values = self._client.eval( 

252 _TRIP_OPEN, 1, self._key(name), _ms(ttl), _fence(expected_version) 

253 ) 

254 return _shared_from_values(values) 

255 

256 def begin_half_open_if_elapsed( 

257 self, *, name: str, wait_duration: float, permitted: int, ttl: float 

258 ) -> SharedState: 

259 """Move ``OPEN`` → ``HALF_OPEN`` once the wait has elapsed (server time).""" 

260 values = self._client.eval( 

261 _BEGIN_HALF_OPEN, 1, self._key(name), wait_duration, permitted, _ms(ttl) 

262 ) 

263 return _shared_from_values(values) 

264 

265 def lease_probe(self, *, name: str, ttl: float) -> ProbeLease: 

266 """Atomically claim one global probe slot.""" 

267 values = self._client.eval(_LEASE_PROBE, 1, self._key(name), _ms(ttl)) 

268 return ProbeLease(granted=bool(int(_s(values[0]))), state=_shared_from_values(values[1:])) 

269 

270 def record_probe(self, *, name: str, outcome: Outcome, ttl: float) -> SharedState: 

271 """Tally one completed probe's outcome (only while ``HALF_OPEN``).""" 

272 values = self._client.eval( 

273 _RECORD_PROBE, 

274 1, 

275 self._key(name), 

276 int(outcome.is_failure), 

277 int(outcome.is_slow), 

278 _ms(ttl), 

279 ) 

280 return _shared_from_values(values) 

281 

282 def close(self, *, name: str, ttl: float, expected_version: int | None = None) -> SharedState: 

283 """Atomically transition to ``CLOSED`` and clear probe accounting. 

284 

285 With ``expected_version``, fenced: a delayed "probes passed" decision 

286 cannot close a breaker that has since re-opened. 

287 """ 

288 values = self._client.eval(_CLOSE, 1, self._key(name), _ms(ttl), _fence(expected_version)) 

289 return _shared_from_values(values)
